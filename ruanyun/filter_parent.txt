protected static function suggestionidValid($suggestionid, $bool=false)
{
    $filter = [];
    $filter['suggestionid'] = $suggestionid;
    $suggestionM = new xxModel();
    $suggestion = $suggestionM -> get($filter);
    if (!$suggestion) {
        Common::setMsgAndCode('suggestionid 参数值非法', ErrorCode::InvalidParam);
    }
    if ($bool) {
        return $suggestion;
    }
    return $suggestionid;
}
